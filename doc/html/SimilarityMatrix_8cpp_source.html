62 int SimilarityMatrix::add(
63  std::string commit1_hash,
64  std::string commit2_hash,
65  double similarity)
66 {
67  /* Check if key already in map, add to end of map if doesn't exist */
68  if (commithashindexmap.count(commit1_hash) == 0)

70  commithashindexmap.insert(

73

78  }
79  mData[getIndex(commit1_hash) * mCols + getIndex(commit2_hash)] = similarity;
80  mData[getIndex(commit2_hash) * mCols + getIndex(commit1_hash)] = similarity;
81

92 std::string SimilarityMatrix::getHash(size_t value)
93 {
94  //https://thispointer.com/how-to-search-by-value-in-a-map-c/
95  std::string hash = "";
96  auto it = commithashindexmap.begin();
97  /* Iterate through the map */
98  while (it != commithashindexmap.end())
99  {
100  /* Check if value of this entry matches with given value */
101  if (it->second == value)
102  {
103  /* Push the key in given map */
104  hash = it->first;
105  return hash;
106  }
107  /* Go to next entry in map */
108  it++;
109  }
110  return hash;
111 }
